## 二、API 调用最佳实践

### 2.1 错误处理与重试

```python
import openai
from tenacity import retry, stop_after_attempt, wait_exponential

@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=1, min=4, max=10),
    retry=retry_if_exception_type((openai.RateLimitError, openai.APIError))
)
def call_llm_with_retry(messages, model="gpt-3.5-turbo"):
    """带重试机制的 LLM 调用"""
    try:
        response = openai.chat.completions.create(
            model=model,
            messages=messages,
            temperature=0.7,
            max_tokens=1000
        )
        return response.choices[0].message.content
    except openai.RateLimitError:
        print("Rate limit hit, retrying...")
        raise
    except Exception as e:
        print(f"Error: {e}")
        raise
```

### 2.2 Token 计算与成本控制

```python
import tiktoken

def count_tokens(text, model="gpt-3.5-turbo"):
    """计算文本的 token 数量"""
    encoding = tiktoken.encoding_for_model(model)
    return len(encoding.encode(text))

def estimate_cost(input_tokens, output_tokens, model="gpt-3.5-turbo"):
    """估算 API 调用成本 (USD)"""
    pricing = {
        "gpt-3.5-turbo": {"input": 0.5, "output": 1.5},  # per 1M tokens
        "gpt-4": {"input": 30, "output": 60},
        "gpt-4-turbo": {"input": 10, "output": 30},
    }
    
    model_pricing = pricing.get(model, pricing["gpt-3.5-turbo"])
    input_cost = (input_tokens / 1_000_000) * model_pricing["input"]
    output_cost = (output_tokens / 1_000_000) * model_pricing["output"]
    
    return input_cost + output_cost
```

## 三、RAG 系统完整实现

### 3.1 文档处理管道

```python
from langchain.document_loaders import PyPDFLoader, TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma
import hashlib

class DocumentProcessor:
    """文档处理管道"""
    
    def __init__(self, chunk_size=500, chunk_overlap=50):
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            length_function=len,
            separators=["\n\n", "\n", "。", " ", ""]
        )
        self.embeddings = OpenAIEmbeddings()
    
    def process_and_store(self, file_path, persist_dir="./chroma_db"):
        """处理文档并存储到向量库"""
        # 加载
        if file_path.endswith('.pdf'):
            loader = PyPDFLoader(file_path)
        else:
            loader = TextLoader(file_path, encoding='utf-8')
        documents = loader.load()
        
        # 分块
        chunks = self.text_splitter.split_documents(documents)
        
        # 添加元数据
        for i, chunk in enumerate(chunks):
            chunk.metadata.update({
                "chunk_id": i,
                "source": file_path,
                "hash": hashlib.md5(chunk.page_content.encode()).hexdigest()[:8]
            })
        
        # 存储
        vectordb = Chroma.from_documents(
            documents=chunks,
            embedding=self.embeddings,
            persist_directory=persist_dir
        )
        vectordb.persist()
        
        return len(chunks)
```

### 3.2 RAG 查询与生成

```python
from langchain.chains import RetrievalQA
from langchain.prompts import PromptTemplate

class RAGSystem:
    """RAG 问答系统"""
    
    def __init__(self, persist_dir="./chroma_db"):
        self.embeddings = OpenAIEmbeddings()
        self.vectordb = Chroma(
            persist_directory=persist_dir,
            embedding_function=self.embeddings
        )
        
        self.prompt_template = """基于以下检索到的信息回答问题。
如果无法从信息中找到答案，请说明"根据现有信息无法回答"。

检索信息：
{context}

问题：{question}

请用中文回答："""
        
        self.PROMPT = PromptTemplate(
            template=self.prompt_template,
            input_variables=["context", "question"]
        )
    
    def query(self, question, k=3):
        """执行 RAG 查询"""
        from langchain.chat_models import ChatOpenAI
        
        llm = ChatOpenAI(temperature=0.3)
        qa_chain = RetrievalQA.from_chain_type(
            llm=llm,
            chain_type="stuff",
            retriever=self.vectordb.as_retriever(search_kwargs={"k": k}),
            return_source_documents=True,
            chain_type_kwargs={"prompt": self.PROMPT}
        )
        
        result = qa_chain({"query": question})
        
        return {
            "answer": result["result"],
            "sources": [doc.metadata for doc in result["source_documents"]]
        }
```

## 五、Agent 开发实战

### 5.1 工具定义

```python
from langchain.tools import BaseTool
from pydantic import BaseModel, Field

class CalculatorInput(BaseModel):
    expression: str = Field(description="数学表达式，如 '2 + 3 * 4'")

class CalculatorTool(BaseTool):
    name = "calculator"
    description = "用于执行数学计算"
    args_schema = CalculatorInput
    
    def _run(self, expression: str):
        try:
            allowed_names = {"abs": abs, "max": max, "min": min}
            result = eval(expression, {"__builtins__": {}}, allowed_names)
            return str(result)
        except Exception as e:
            return f"计算错误: {e}"
```

### 5.2 ReAct Agent

```python
from langchain.agents import initialize_agent, AgentType
from langchain.memory import ConversationBufferMemory

def create_agent(tools, llm):
    """创建 ReAct Agent"""
    memory = ConversationBufferMemory(
        memory_key="chat_history",
        return_messages=True
    )
    
    agent = initialize_agent(
        tools=tools,
        llm=llm,
        agent=AgentType.CHAT_CONVERSATIONAL_REACT_DESCRIPTION,
        memory=memory,
        verbose=True,
        handle_parsing_errors=True
    )
    
    return agent
```

### 6.2 日志记录

```python
import logging
import json
from datetime import datetime

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('llm_app.log'),
        logging.StreamHandler()
    ]
)

def log_llm_call(model, input_tokens, output_tokens, latency, success):
    """记录 LLM 调用日志"""
    log_data = {
        "timestamp": datetime.utcnow().isoformat(),
        "model": model,
        "input_tokens": input_tokens,
        "output_tokens": output_tokens,
        "latency_ms": latency,
        "success": success,
        "cost_usd": estimate_cost(input_tokens, output_tokens, model)
    }
    logging.info(f"LLM_CALL: {json.dumps(log_data)}")
```
